107 GLfloat FindRotationAngle(glm::vec3 startVec, glm::vec3 endVec) {
108  GLfloat angle, zA, zB, xA, xB, dotProd;
109  GLfloat width = WIN_WIDTH / 2.0f;
110 
111  xA = glm::sin(((startVec.x - width) / width) * PI / 2);
112  xB = glm::sin(((endVec.x - width) / width) * PI / 2);
113 
114  zA = glm::sqrt(1.0f - (xA * xA));
115  zB = glm::sqrt(1.0f - (xB * xB));
116 
117  glm::vec3 vA(xA, 0.0, zA);
118  glm::vec3 vB(xB, 0.0, zB);
119 
120  dotProd = glm::dot(glm::normalize(vA), glm::normalize(vB));
121 
122  return glm::acos(dotProd) * 1.20f;
123 }
